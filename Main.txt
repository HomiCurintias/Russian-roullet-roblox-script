local function startCrazyFly()
	local char = player.Character or player.CharacterAdded:Wait()
	local hrp = char:WaitForChild("HumanoidRootPart")
	local RunService = game:GetService("RunService")

	-- Loop de movimento absurdo
	RunService.RenderStepped:Connect(function()
		if not hrp or not hrp.Parent then return end

		local direction = Vector3.new(
			math.random(-1, 1),
			math.random(0, 1),
			math.random(-1, 1)
	 ).Unit

		hrp.Velocity = direction * 999999
	end)
end

local function sla() 
    local Rng = math.random(0, 1)
    if Rng == 0 then
        local character = player.Character
        if character and character:FindFirstChild("Humanoid") then
            character.Humanoid.WalkSpeed = character.Humanoid.WalkSpeed + 10
        end
    else
        print("Você perdeu. Entrando em modo loucura...")
        startCrazyFly()
    end
end

but.MouseButton1Click:Connect(sla)
